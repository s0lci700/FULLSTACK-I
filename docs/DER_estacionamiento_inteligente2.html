monto_base = precio_base_hora × multiplicador_horario × factor_tipo_vehiculo × factor_tipo_espacio × (minutos / 60)
monto_final = monto_base × (1 − desc_tipo_cliente) × (1 − desc_suscripcion) × (1 − desc_banco)
