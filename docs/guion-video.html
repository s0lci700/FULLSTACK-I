public String login(LoginRequestDTO dto) {
    log.info("Intentando iniciar sesión con email: {}", dto.getEmail());
    UserCredential credencial = userCredentialRepository
        .findByEmail(dto.getEmail())               // busca por email, no por username
        .orElseThrow(() -> new InvalidCredentialsException("Credenciales inválidas"));
    if (!passwordEncoder.matches(dto.getPassword(), credencial.getPasswordHash())) {
        throw new InvalidCredentialsException("Credenciales inválidas");
    }
    return jwtUtil.generateToken(credencial.getEmail(), credencial.getRol().getNombre());
}

@Transactional
public ReservaResponseDTO create(ReservaCreateDTO reserva) {
    // 1. Validar cliente via Feign → user-service
    try { cliente = clienteClient.findById(reserva.getIdCliente()); }
    catch (FeignException.NotFound e) { throw new NotFoundException("Cliente no encontrado"); }
    if (!cliente.getActivo()) throw new NotFoundException("Cliente inactivo");

    // 2. Validar vehículo via Feign → ms-vehiculos
    try { vehiculo = vehiculoClient.findById(reserva.getIdVehiculo()); }
    catch (FeignException.NotFound e) { throw new NotFoundException("Vehículo no encontrado"); }

    // 3. Validar espacio via Feign → ms-espacios
    try { espacio = espacioClient.findById(reserva.getIdEspacio()); }
    catch (FeignException.NotFound e) { throw new NotFoundException("Espacio no encontrado"); }
    if (!espacio.getActivo() || !espacio.getDisponible())
        throw new NotFoundException("Espacio inactivo o no disponible");

    nReserva.setEstado(EstadoEnums.PENDIENTE);   // siempre empieza PENDIENTE
    return toDTO(reservasRepository.save(nReserva));
}

@Transactional
public ReservaResponseDTO confirmar(Long id) {
    Reserva reserva = reservasRepository.findById(id)
        .orElseThrow(() -> new NotFoundException("Reserva no encontrada"));
    if (reserva.getEstado() == EstadoEnums.CONFIRMADA || ...)
        throw new ConflictException("No se puede confirmar en estado actual");
    reserva.setEstado(EstadoEnums.CONFIRMADA);
    espacioClient.updateDisponibilidad(reserva.getIdEspacio(), false);  // bloquear espacio
    return toDTO(reservasRepository.save(reserva));
}

@Transactional
public AccesoResponseDTO registrarSalida(Long id) {
    Acceso acceso = accesoRepository.findById(id)
        .orElseThrow(() -> new NotFoundException("Acceso no encontrado"));
    if (acceso.getFechaHoraSalida() != null)
        throw new ConflictException("Salida ya registrada para este acceso");

    long minutos = ChronoUnit.MINUTES.between(acceso.getFechaHoraEntrada(), LocalDateTime.now());
    acceso.setFechaHoraSalida(LocalDateTime.now());
    acceso.setMinutos((int) minutos);
    acceso.setEstado(EstadoEnum.COMPLETADO);

    espacioClient.updateDisponibilidad(acceso.getIdEspacio(), true);  // liberar espacio
    reservaClient.finalizarReserva(acceso.getIdReserva());            // → FINALIZADA
    return toDTO(accesoRepository.save(acceso));
}

// monto_base = precio_hora × mult_horario × factor_veh × factor_esp × (min/60)
private BigDecimal calcularMontoBase(TarifaResponseDTO tarifa,
        BigDecimal factorVehiculo, BigDecimal factorEspacio,
        BigDecimal multiplicadorHorario, int minutos) {
    BigDecimal horas = BigDecimal.valueOf(minutos)
        .divide(BigDecimal.valueOf(60), 4, RoundingMode.HALF_UP);
    return BigDecimal.valueOf(tarifa.getPrecioBaseHora().doubleValue())
        .multiply(multiplicadorHorario).multiply(factorVehiculo)
        .multiply(factorEspacio).multiply(horas)
        .setScale(2, RoundingMode.HALF_UP);
}

// monto_final = monto_base × (1 - desc_cliente) × (1 - desc_suscripcion) × (1 - desc_banco)
private BigDecimal calcularMontoFinal(BigDecimal montoBase,
        BigDecimal descCliente, BigDecimal descSuscripcion, BigDecimal descBanco) {
    BigDecimal cien = BigDecimal.valueOf(100);
    return montoBase
        .multiply(BigDecimal.ONE.subtract(descCliente.divide(cien, 4, RoundingMode.HALF_UP)))
        .multiply(BigDecimal.ONE.subtract(descSuscripcion.divide(cien, 4, RoundingMode.HALF_UP)))
        .multiply(BigDecimal.ONE.subtract(descBanco.divide(cien, 4, RoundingMode.HALF_UP)))
        .setScale(2, RoundingMode.HALF_UP);
}

@ExtendWith(MockitoExtension.class)
class AccesoServiceTest {

    @Mock private AccesoRepository  accesoRepository;
    @Mock private EspacioClient     espacioClient;   // Feign mockeado — sin red
    @Mock private ReservaClient     reservaClient;   // Feign mockeado — sin Eureka
    @InjectMocks private AccesoService accesoService;

    @Test
    @DisplayName("registrarSalida debe calcular minutos, cambiar estado a COMPLETADO y liberar espacio")
    void registrarSalida_valido_debeCalcularMinutosYLiberarEspacio() {
        // Arrange
        when(accesoRepository.findById(1L)).thenReturn(Optional.of(accesoActivo));
        when(accesoRepository.save(any(Acceso.class))).thenAnswer(inv -> inv.getArgument(0));

        // Act
        AccesoResponseDTO resultado = accesoService.registrarSalida(1L);

        // Assert
        assertThat(resultado.getEstado()).isEqualTo("COMPLETADO");
        assertThat(resultado.getMinutos()).isGreaterThan(0L);
        verify(espacioClient).updateDisponibilidad(1L, true);  // espacio liberado
        verify(reservaClient).finalizarReserva(1L);           // reserva finalizada
    }
}

@Slf4j
@RestControllerAdvice   // intercepta excepciones de TODOS los controllers automáticamente
public class GlobalExceptionHandler {

    private Map<String,Object> construirRespuesta(HttpStatus status, String mensaje) {
        // body uniforme: { timestamp, status, mensaje }
        Map<String,Object> r = new HashMap<>();
        r.put("timestamp", LocalDateTime.now()); r.put("status", status.value());
        r.put("mensaje", mensaje); return r;
    }

    @ExceptionHandler(NotFoundException.class)              // → HTTP 404
    public ResponseEntity<...> manejarNoEncontrado(NotFoundException ex) { ... }

    @ExceptionHandler(ConflictException.class)             // → HTTP 409
    public ResponseEntity<...> manejarConflicto(ConflictException ex) { ... }

    @ExceptionHandler(MethodArgumentNotValidException.class) // → HTTP 400 + campos
    public ResponseEntity<...> manejarValidaciones(MethodArgumentNotValidException ex) {
        // agrega mapa "errores": {"campo": "mensaje de error"}
    }
    // + BadRequestException → 400, BusinessException → 422
}
